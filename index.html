@app.route('/api/v1/process-resume', methods=['POST'])
def process_resume():
    """
    Process uploaded resume and extract information
    
    Request:
    --------
    - Method: POST
    - Content-Type: multipart/form-data
    - Body: 
        - file: PDF file (required)
        - user_id: string (required)
    
    Response:
    ---------
    - 200: Success
        {
            "status": "success",
            "data": {
                "skills": [...],
                "experience": {...},
                "education": [...]
            }
        }
    - 400: Bad Request
    - 500: Server Error
    """
    try:
        # Validate request
        if 'file' not in request.files:
            return jsonify({"error": "No file provided"}), 400
            
        file = request.files['file']
        
        # Process PDF using PDFMiner
        text = process_pdf(file)
        
        # Extract information using NLP
        data = extract_information(text)
        
        return jsonify({
            "status": "success",
            "data": data
        }), 200
        
    except Exception as e:
        return jsonify({"error": str(e)}), 500

2. Skill Extraction API


# Skill Extraction Service
class SkillExtractor:
    """
    Service to extract and analyze skills from resume text
    
    Methods:
    --------
    extract_skills(text: str) -> List[str]
    match_skills(skills: List[str]) -> Dict[str, float]
    """
    
    def __init__(self):
        """Initialize NLP models and skill database"""
        self.nlp = spacy.load("en_core_web_sm")
        self.skill_patterns = load_skill_patterns()
    
    def extract_skills(self, text: str) -> List[str]:
        """
        Extract skills from text using NLP
        
        Parameters:
        -----------
        text : str
            Raw text from resume
            
        Returns:
        --------
        List[str]
            List of identified skills
        """
        doc = self.nlp(text)
        skills = []
        
        # Pattern matching for technical skills
        for pattern in self.skill_patterns:
            if pattern.search(text):
                skills.append(pattern.group())
                
        # Named Entity Recognition for additional skills
        for ent in doc.ents:
            if ent.label_ == "SKILL":
                skills.append(ent.text)
                
        return list(set(skills))  # Remove duplicates

# Course Recommendation Engine
class CourseRecommender:
    """
    Recommends relevant courses based on extracted skills
    
    Methods:
    --------
    get_recommendations(skills: List[str], limit: int = 5) -> List[Dict]
    """
    
    def __init__(self):
        """Initialize course database and matching algorithm"""
        self.courses_db = load_courses_database()
        self.skill_matcher = SkillMatcher()
    
    def get_recommendations(
        self, 
        skills: List[str], 
        limit: int = 5
    ) -> List[Dict]:
        """
        Get course recommendations based on skills
        
        Parameters:
        -----------
        skills : List[str]
            List of user's current skills
        limit : int, optional
            Maximum number of recommendations
            
        Returns:
        --------
        List[Dict]
            Recommended courses with relevance scores
        """
        recommendations = []
        
        # Match skills with courses
        for course in self.courses_db:
            score = self.skill_matcher.calculate_relevance(
                skills, 
                course.required_skills
            )
            if score > 0.5:  # Minimum relevance threshold
                recommendations.append({
                    "course_id": course.id,
                    "title": course.title,
                    "relevance": score,
                    "url": course.url
                })
        
        # Sort by relevance and limit results
        recommendations.sort(key=lambda x: x["relevance"], reverse=True)
        return recommendations[:limit]

# Error Handler
class ErrorHandler:
    """
    Centralized error handling for the application
    
    Methods:
    --------
    handle_error(error: Exception) -> Tuple[Dict, int]
    log_error(error: Exception, context: Dict) -> None
    """
    
    def handle_error(self, error: Exception) -> Tuple[Dict, int]:
        """
        Handle and format error responses
        
        Parameters:
        -----------
        error : Exception
            The caught exception
            
        Returns:
        --------
        Tuple[Dict, int]
            Error response and HTTP status code
        """
        if isinstance(error, ValidationError):
            return {
                "error": "Validation Error",
                "details": error.messages
            }, 400
            
        elif isinstance(error, AuthenticationError):
            return {
                "error": "Authentication Error",
                "message": str(error)
            }, 401
            
        elif isinstance(error, DatabaseError):
            return {
                "error": "Database Error",
                "message": "Internal server error"
            }, 500
            
        # Log unexpected errors
        self.log_error(error, {
            "timestamp": datetime.now(),
            "type": type(error).__name__
        })

# Rate Limiter
class RateLimiter:
    """
    Implements rate limiting for API endpoints
    
    Methods:
    --------
    check_rate_limit(user_id: str) -> bool
    update_request_count(user_id: str) -> None
    """
    
    def __init__(self):
        """Initialize rate limiter with Redis"""
        self.redis_client = Redis(host='localhost', port=6379)
        self.rate_limit = 100  # requests per hour
        self.window = 3600  # 1 hour in seconds
    
    def check_rate_limit(self, user_id: str) -> bool:
        """
        Check if user has exceeded rate limit
        
        Parameters:
        -----------
        user_id : str
            Unique identifier for the user
            
        Returns:
        --------
        bool
            True if under limit, False if exceeded
        """
        key = f"rate_limit:{user_id}"
        current = int(self.redis_client.get(key) or 0)
        
        if current >= self.rate_limit:
            return False
            
        # Update counter in Redis
        pipe = self.redis_client.pipeline()
        pipe.incr(key)
        pipe.expire(key, self.window)
        pipe.execute()
        
        return True
